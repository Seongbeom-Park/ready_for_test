# BFS
```c
int curr[MAX], curr_len = 0;
int next[MAX], next_len = 0;

while (curr_len > 0) {
  next_len = 0;
  for (next_id) { // 다음에 방문 할 node들을 next에 넣기
    next[next_len] = next_id;
    next_len++;
  }
  for (i=0; i<next_len; i++) {
    curr[i] = next[i]; // 리스트에 들어있는 요소 옮기기
  }
  curr_len = next_len; // 리스트 크기 바꾸기
}
```
